create view v_unit_states as
select
   ccc.id        as id,
   ccc.name      as name,
   ccc.flag1     as flag1,
   ccc.filename1 as filename1,
   ccc.filename2 as filename2,
   ccc.filename3 as filename3,
   img.name      as filename4
from
(select
   bbb.id        as id,
   bbb.name      as name,
   bbb.flag1     as flag1,
   bbb.back1     as back1,
   bbb.filename1 as filename1,
   bbb.filename2 as filename2,
   img.name      as filename3
from
(select
   aaa.id        as id,
   aaa.name      as name,
   aaa.flag1     as flag1,
   aaa.front1    as front1,
   aaa.back1     as back1,
   aaa.filename1 as filename1,
   img.name      as filename2
from 
(select
   unit.id      as id,
   unit.name    as name,
   unit.back    as back,
   unit.flag1   as flag1,
   unit.front1  as front1,
   unit.back1   as back1,
   img.name     as filename1
from unit, img
where
unit.front = img.id) as aaa,
img
where
aaa.back = img.id) as bbb,
img
where
bbb.front1 = img.id) as ccc,
img
where
ccc.back1 = img.id;
