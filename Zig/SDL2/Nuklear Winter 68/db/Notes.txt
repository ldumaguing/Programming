create view v_chits as
select
cc.id, cc.name, cc.frontname, ii.name backname, cc.unit_type, cc.formation, cc.flags
from
images ii,
(select 
c.id, c.name, i.name frontname, c.back, c.unit_type, c.formation, c.flags
from
  images i,
  chits  c
where
c.front = i.id
and
c.is_chit = 1
) cc
where
cc.back = ii.id
;
