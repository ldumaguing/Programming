cmake_minimum_required(VERSION 3.22)

include(/opt/pico-sdk/external/pico_sdk_import.cmake)

set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)
set(PROJ_NAME fish)

project(${PROJ_NAME} C CXX ASM)

pico_sdk_init()






add_library(TFTLCD
	stuffs/TFTLCD-Library/MAGA_TFTLCD.cpp
)
target_link_libraries(TFTLCD pico_stdlib)
target_include_directories(TFTLCD PUBLIC stuffs)

add_library(GFX
	stuffs/MAGA-GFX-Library/MAGA_GFX.cpp
)
target_link_libraries(GFX TFTLCD pico_stdlib)
target_include_directories(GFX PUBLIC stuffs)






add_executable(${PROJ_NAME}
	graphicstest.cpp
)




pico_enable_stdio_usb(${PROJ_NAME} 1)
pico_add_extra_outputs(${PROJ_NAME})

target_link_libraries(${PROJ_NAME}
	GFX
	pico_stdlib
	pico_mem_ops
)
