CREATE TABLE card (
    id    INTEGER PRIMARY KEY,
    file  TEXT,
    flag1 INTEGER DEFAULT 0
);

flag1
=====
bit 0: character
bit 1: blue deck (Vampire)
bit 2: cyan deck (Lycan)
bit 3: fuchsia deck (Witch)
bit 4: green deck (Zombie)
bit 5: purple deck
bit 6: white deck (basic)
bit 7: rear deck image


update card set flag1 = (flag1 | (1<<7))
where id in (
1030,1041,1043,1053,1064
);



alter table card add column id integer;





.headers on
.mode csv
.output card.csv
select id, file, 0 from card order by id;
.output stdout
.quit

.mode csv
.import card.csv card
