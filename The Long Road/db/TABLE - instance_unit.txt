CREATE TABLE instance_unit (
    id          INTEGER,
    scenario_id INTEGER NOT NULL,     -- "foreign key" from _scenario_ table
    unit_id     INTEGER NOT NULL,
    element     TEXT,                 -- sub faction/group
    loc_x       INTEGER DEFAULT 0,
    loc_y       INTEGER DEFAULT 0,
    loc_z       INTEGER DEFAULT 0,
    stack_lvl   INTEGER DEFAULT 0,
    status1     INTEGER DEFAULT 0,
    status2     INTEGER DEFAULT 0,
    -- ****************************** data from unit
    descrip   TEXT,
    chit_id1  INTEGER DEFAULT 0,
    chit_id2  INTEGER DEFAULT 0,
    faction   TEXT,
    "flag1:u" INTEGER DEFAULT 0,
    -- ****************************** data from chit
    chit_id INTEGER,
    front   INTEGER,
    back    INTEGER,
    health  INTEGER,  -- info for champions
    flip    INTEGER   -- info for champions
    -- ****************************** data from img
    img_id     INTEGER DEFAULT 0,
    file       TEXT,
    flag1      INTEGER DEFAULT 0,
    apf_val    INTEGER DEFAULT -100,
    apf_rng    TEXT,
    hef_val    INTEGER DEFAULT -100,
    hef_rng    TEXT,
    mf_val     INTEGER DEFAULT -100,
    caf_val    INTEGER DEFAULT -100,
    flag2      INTEGER DEFAULT 0,
    armor_val  INTEGER DEFAULT -100,
    flag3      INTEGER DEFAULT 0,
    ability1   INTEGER DEFAULT 0,
    ability2   INTEGER DEFAULT 0,
    apf2_val   INTEGER DEFAULT -100,
    apf2_rng   TEXT,
    hef2_val   INTEGER DEFAULT -100,
    hef2_rng   TEXT,
    URPapf_val TEXT, -- Userper's two values
    URPhef_val TEXT, -- Userper's two values
    ability3   INTEGER DEFAULT 0,
    --
    PRIMARY KEY (id, scenario_id)
);

flag1
=====
number of img states - bit 1,0:
   00: two-sided
   01: one-sided
   10: three-sided
   11: four-sided
4-img units - bit 2-4:
   000: default 2-img units
   001: VTOL
   010: Transformer
   011: Reborn
   100: 4-HP units

status2
=======
current chit side - 0,1:
   00: 1st chit, front
   01: 1st chit, back
   10: 2nd chit, front
   11: 2nd chit, back

status1
=======
bit  0: invisible
bit  1: -
bit  2: hovering
bit  3: werewolf
bit  4: AFV using wired missile
bit  5: with Songster (T-64B) or use a unit stat/img with a missile icon
bit  6: missile used up
bit  7: fired
bit  8: moved
bit  9: disrupted
bit 10: has infintry/recon
bit 11: has hero 1
bit 12: has hero 2
facing 15-13:
   000: N
   001: NE
   010: SE
   011: S
   100: SW
   101: NW
