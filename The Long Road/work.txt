select
   bbb.id,
   bbb.descrip,
   bbb.A_front,
   bbb.A_back,
   img.file,
   bbb.B_back
from (
select
   aaa.id      id,
   aaa.descrip descrip,
   aaa.A_front A_front,
   img.file    A_back,
   aaa.B_front B_front,
   aaa.B_back  B_back
from (
select
   mv_unit.id       id,
   mv_unit.descrip  descrip,
   img.file         A_front,
   mv_unit.A_back   A_back,
   mv_unit.B_front  B_front,
   mv_unit.B_back   B_back
from mv_unit
left join img
on
mv_unit.A_front = img.id) aaa
left join img
on
aaa.A_back = img.id) bbb
left join img
on
bbb.B_front = img.id
;

CREATE TABLE mv_unit(
  id INT,
  descrip TEXT,
  A_front INT,
  A_back INT,
  B_front INT,
  B_back INT,
  faction TEXT,
  version INT,
  flag1 INT
);


























select
   unit.id       id,
   unit.descrip  descrip,
   unit.chit_id2 chit_id2,
   chit.front    A_front,
   chit.back     A_back,
   unit.faction  faction,
   unit.version  version,
   unit.flag1    flag1
from
unit, chit
where
unit.chit_id1 = chit.id
;










create view v_unit as
select
   aaa.id      id,
   aaa.descrip descrip,
   aaa.A_front A_front,
   aaa.A_back  A_back,
   chit.front  B_front,
   chit.back   B_back,
   aaa.faction faction,
   aaa.version version,
   aaa.flag1   flag1
from
(
select
   unit.id       id,
   unit.descrip  descrip,
   unit.chit_id2 chit_id2,
   chit.front    A_front,
   chit.back     A_back,
   unit.faction  faction,
   unit.version  version,
   unit.flag1    flag1
from
unit, chit
where
unit.chit_id1 = chit.id
) aaa
left join chit
on
aaa.chit_id2 = chit.id;































CREATE TABLE unit (
    id       INTEGER PRIMARY KEY,
    descrip  TEXT,
    chit_id1 INTEGER DEFAULT 0,
    chit_id2 INTEGER DEFAULT 0,
    faction  TEXT,
    version  INTEGER DEFAULT 0,
    flag1    INTEGER DEFAULT 0
);


CREATE TABLE chit (
    id      INTEGER,
    descrip TEXT,
    front   INTEGER,
    back    INTEGER,
    faction TEXT,
    version INTEGER
);
