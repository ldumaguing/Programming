select hexID, loc_x, loc_y,
   flag1 & 57345,        -- hill, rolling, forest, town
   (flag1 & 16510) >> 1,                                                 -- road
   ((((flag1 & 16510) >> 1) >> 3) | (((flag1 & 16510) >> 1) << 3)) & 63, -- rotate road
   (flag1 & 8064) >> 7                                                   -- tunnel
from terrain
where
flag1 > 0
and
mapFile = 'Map A';

and
hexID = 'K11';
